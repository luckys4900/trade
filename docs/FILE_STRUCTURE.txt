┌─ 【自動生成ファイル - システムが自動で作成・更新】
│
├─ whale_signal.json
│  │ 役割: クジラシグナル（最も重要）
│  │ 生成元: whale_monitor.py
│  │ 更新時刻: 15分ごと
│  │ 内容例:
│  │  {
│  │    "direction": "NONE",
│  │    "valid": false,
│  │    "strength": 0.0,
│  │    "wallet_count": 3,
│  │    "n_ranked": 0,
│  │    "timestamp": 1775836438662
│  │  }
│  │ 見方:
│  │  • "valid": true  → シグナル有効（取引チャンス！）
│  │  • "valid": false → シグナルなし（様子見）
│  │  • "direction": "LONG" → 買いシグナル
│  │  • "direction": "SHORT" → 売りシグナル
│  │  • "strength": 大きいほど確実性が高い
│
├─ macro_state.json
│  │ 役割: 市場環境フィルター
│  │ 生成元: macro_filter.py
│  │ 更新時刻: 60分ごと
│  │ 内容例:
│  │  {
│  │    "regime": "NORMAL",
│  │    "atr_ratio": 0.0150,
│  │    "caution_mode": false,
│  │    "next_event": "2026-04-15 14:30:00",
│  │    "hours_to_event": 125.7
│  │  }
│  │ 見方:
│  │  • "regime": "NORMAL" → 通常モード（取引実行可）
│  │  • "regime": "EXTREME" → 経済指標周辺（取引スキップ）
│  │  • "caution_mode": false → 安全
│  │  • "caution_mode": true → ボラティリティ高い（注意）
│
